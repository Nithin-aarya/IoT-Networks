/*
 * ESP32 #1 - Access Point
 * Creates Wi-Fi network and allows dynamic channel changes via HTTP.
 */

#include <WiFi.h>
#include <WebServer.h>

const char* ssid = "DynamicAP";
const char* password = "password";

// Static IP configuration
IPAddress local_ip(192,168,4,1);
IPAddress gateway(192,168,4,1);
IPAddress subnet(255,255,255,0);

WebServer server(80);

// -------------------------------------------------------------------
void handleRoot() {
  server.send(200, "text/plain", "AP is running");
}

// -------------------------------------------------------------------
// Handler to change Wi-Fi channel
// In ESP32_AP.ino, replace handleSetChannel() with:

void handleSetChannel() {
  if (server.hasArg("channel")) {
    int channel = server.arg("channel").toInt();
    if (channel >= 1 && channel <= 11) {          // Only allow 1-11
      bool success = WiFi.setChannel(channel);
      if (success) {
        server.send(200, "text/plain", "OK, channel set to " + String(channel));
        Serial.print("Channel changed to ");
        Serial.println(channel);
      } else {
        server.send(500, "text/plain", "Failed to set channel");
        Serial.println("Channel change failed");
      }
    } else {
      server.send(400, "text/plain", "Channel must be 1-11");
    }
  } else {
    server.send(400, "text/plain", "Missing channel");
  }
}
// -------------------------------------------------------------------
// Handler to report current channel
void handleStatus() {
  server.send(200, "text/plain", "Channel: " + String(WiFi.channel()));
}

// -------------------------------------------------------------------
void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_AP);
  WiFi.softAPConfig(local_ip, gateway, subnet);
  WiFi.softAP(ssid, password, 1);          // Start on channel 1
  server.on("/", handleRoot);
  server.on("/setchannel", handleSetChannel);
  server.on("/status", handleStatus);
  server.begin();
  Serial.println("AP started on channel 1");
}

// -------------------------------------------------------------------
void loop() {
  server.handleClient();
}
